من نحن

اهلا بكم في ميال

نشــأت شــركة الميــال للمقــاولات بهــدف اعتــاء موقــع مرمــوق فــي الســوق الســعودي ولتكــون شــركة رائــدة فــي مجــال الأعمــال الكهربائيـة وخاصـة فـي مجـال نقـل الطاقـة و شـبكات التوزيـع للجهـد المتوسـط والمخفـض وقـد عـزز مـن ذلـك التأهيـل المبكـر لنـا مـع الشـركة السـعودية للكهربـاء بالعمـل كمقـاول معتمـد لديهـا فـي مجـال أعمـال التوزيـع ونقـل الطاقـة . وسـاعد الانتشـار الجغرافـي للشـركة فـي أنحـاء المملكـة فـي أن تأخـذ الشـركة المكانـة المميـزة لهـا.

بداية الشركة

نشـأت الشـركة منـذ 1413/11/27هــ - 1993/5/19م وخـال هـذه الفتـرة اسـتطعنا إثبـات جدارتنـا ومقدرتنـا مـن خــال عمــل احترافــي مبنــي علــى الجــودة والســرعة. بــدأت الشــركة أعمالهــا فــي مدينــة جــدة وامتــدت فروعهــا إلــى الريـاض والمدينـة المنـورة ومكـة المكرمـة وينبـع و مهـد الدهـب ونسـعى للتوسـع فـي جميـع أنحـاء المملكـة وخـال هـذه الفتــرة قامــت بتنفيــذ وإنشــاء وتشــغيل وصيانــة مشــاريع الأعمــال الكهربائيــة الخاصــة لشــبكات الجهــد العالــي والمتوسـط والمنخفـض والمسـاهمة فـي مجـال التطويـر العقـاري والاسـتثمار لتحقـق بذلـك نقلـة نوعيـة فـي طريقـة عملهـا مـن حيـث الحجـم و أسـلوب العمـل .

الرسالة والرؤية

الرسالة

مواكبــة أحــدث التقنيــات التــي تشــهدها الاعمــال الكهربائيــة الخاصــة لشــبكات الجهــد العالــي والمتوســط والمنخفــض والمشــاريع الفنيــة والتقنيــة علــى المســتوى الوطنــي والإقليمــي والعالمــي، ويشـمل ذلـك تطويـر وتدريـب وتأهيـل الكـوادر الفنيـة المؤهلـة، واسـتخدام أحـدث الأجهـزة والمعـدات والآليــات التــي تضمــن تقديــم أفضــل الخدمــات الفنيــة والإداريــة ومواصلــة رفــع مســتوى الأداء وفــق مــع مــا تشــهده هــذه الصناعــة مــن تطــور ً معاييــر ومؤشــرات قيــاس الأداء

الرؤية

نؤمـن تمامـا بضـرورة تحقيـق مسـتوى الجـودة الشـاملة والمتكاملـة فـي جميـع المشـار ً يع وذلـك إيمانـا بـأن ذلـك هـو السـبيل لتحقيـق النجـاح

شركاء النجاح

الشــركة أعمالهــا فــي مدينــة جــدة وامتــدت فروعهــا إلــى الريـاض والمدينـة المنـورة ومكـة المكرمـة وينبـع و مهـد الدهـب ونسـعى للتوسـع فـي جميـع أنحـاء المملكـة
